Generate a random number between 50 and 100 inclusively.

function getRndNumber(min, max) {
    let numberRnd = Math.floor(Math.random() * (max - min + 1)) + min;
    return numberRnd;
}
let getNumber = getRndNumber(50, 100);
console.log(getNumber);
